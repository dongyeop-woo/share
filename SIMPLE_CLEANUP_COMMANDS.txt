# SSH에서 직접 타이핑할 간단한 명령어들
# 한 줄씩 복사해서 타이핑하거나, 마우스 오른쪽 클릭으로 붙여넣기 시도

# 1단계: 로그 정리
sudo journalctl --vacuum-time=1d

# 2단계: apt 정리
sudo apt-get clean
sudo apt-get autoremove -y

# 3단계: 임시 파일 삭제
sudo rm -rf /tmp/*

# 4단계: 기존 서비스 중지
sudo systemctl stop share-frontend 2>/dev/null
sudo systemctl stop share-backend 2>/dev/null
sudo pkill -f python

# 5단계: 기존 프로젝트 파일 삭제
sudo rm -rf /var/www/share
sudo rm -rf /opt/share-backend

# 6단계: 공간 확인
df -h
